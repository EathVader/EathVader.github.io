2 四种遍历 / Traversals

前序遍历（递归 + 迭代）

// 递归
void preorder(TreeNode root, List<Integer> res) {
    if (root == null) return;
    res.add(root.val);        // 根
    preorder(root.left, res); // 左
    preorder(root.right, res);// 右
}

// 迭代（用栈模拟）
List<Integer> preorderIterative(TreeNode root) {
    List<Integer> res = new ArrayList<>();
    Stack<TreeNode> stack = new Stack<>();
    if (root != null) stack.push(root);
    while (!stack.isEmpty()) {
        TreeNode node = stack.pop();
        res.add(node.val);
        if (node.right != null) stack.push(node.right); // 先右后左
        if (node.left != null) stack.push(node.left);
    }
    return res;
}

层序遍历

List<List<Integer>> levelOrder(TreeNode root) {
    List<List<Integer>> res = new ArrayList<>();
    if (root == null) return res;
    Queue<TreeNode> queue = new LinkedList<>();
    queue.offer(root);
    while (!queue.isEmpty()) {
        int size = queue.size();
        List<Integer> level = new ArrayList<>();
        for (int i = 0; i < size; i++) {
            TreeNode node = queue.poll();
            level.add(node.val);
            if (node.left != null) queue.offer(node.left);
            if (node.right != null) queue.offer(node.right);
        }
        res.add(level);
    }
    return res;
}

3 深度与节点 / Depth & Nodes

① 最大深度

int maxDepth(TreeNode root) {
    if (root == null) return 0;
    return Math.max(maxDepth(root.left), maxDepth(root.right)) + 1;
}

② 最小深度

⚠️ 陷阱

最小深度是到最近叶节点的距离。如果一侧子树为空，不能返回 0，要走另一侧。

int minDepth(TreeNode root) {
    if (root == null) return 0;
    if (root.left == null) return minDepth(root.right) + 1;
    if (root.right == null) return minDepth(root.left) + 1;
    return Math.min(minDepth(root.left), minDepth(root.right)) + 1;
}

③ 节点总数 / ④ 叶子节点数 / ⑤ 第 K 层节点数

// 节点总数
int countNodes(TreeNode root) {
    if (root == null) return 0;
    return countNodes(root.left) + countNodes(root.right) + 1;
}

// 叶子节点数
int countLeaves(TreeNode root) {
    if (root == null) return 0;
    if (root.left == null && root.right == null) return 1;
    return countLeaves(root.left) + countLeaves(root.right);
}

// 第 K 层节点数
int countAtLevel(TreeNode root, int k) {
    if (root == null || k < 1) return 0;
    if (k == 1) return 1;
    return countAtLevel(root.left, k-1) + countAtLevel(root.right, k-1);
}
💡 递归三要素

所有这些题都遵循同一模式：① base case（null 返回）→ ② 递归左右子树 → ③ 合并结果。掌握这个模式，大部分二叉树题都能秒解。

4 结构判断 / Structure Checks

⑥ 是否平衡二叉树

boolean isBalanced(TreeNode root) {
    return height(root) != -1;
}
int height(TreeNode root) {
    if (root == null) return 0;
    int l = height(root.left), r = height(root.right);
    if (l == -1 || r == -1 || Math.abs(l - r) > 1) return -1;
    return Math.max(l, r) + 1;
}

⑦ 是否完全二叉树

// BFS：遇到 null 后不应再有非 null 节点
boolean isComplete(TreeNode root) {
    if (root == null) return true;
    Queue<TreeNode> q = new LinkedList<>();
    q.offer(root);
    boolean seenNull = false;
    while (!q.isEmpty()) {
        TreeNode node = q.poll();
        if (node == null) { seenNull = true; continue; }
        if (seenNull) return false;
        q.offer(node.left);
        q.offer(node.right);
    }
    return true;
}

⑧ 两棵树是否相同 / ⑨ 是否镜像

// 相同
boolean isSame(TreeNode p, TreeNode q) {
    if (p == null && q == null) return true;
    if (p == null || q == null) return false;
    return p.val == q.val && isSame(p.left, q.left) && isSame(p.right, q.right);
}

// 镜像
boolean isMirror(TreeNode p, TreeNode q) {
    if (p == null && q == null) return true;
    if (p == null || q == null) return false;
    return p.val == q.val && isMirror(p.left, q.right) && isMirror(p.right, q.left);
}

5 变换操作 / Transformations

⑩ 翻转二叉树（镜像）

TreeNode invertTree(TreeNode root) {
    if (root == null) return null;
    TreeNode tmp = root.left;
    root.left = invertTree(root.right);
    root.right = invertTree(tmp);
    return root;
}

⑪ 最低公共祖先 (LCA)

TreeNode lowestCommonAncestor(TreeNode root, TreeNode p, TreeNode q) {
    if (root == null || root == p || root == q) return root;
    TreeNode left = lowestCommonAncestor(root.left, p, q);
    TreeNode right = lowestCommonAncestor(root.right, p, q);
    if (left != null && right != null) return root; // p, q 分居两侧
    return left != null ? left : right;
}

⑫ 从前序 + 中序构建二叉树

TreeNode buildTree(int[] pre, int[] in) {
    Map<Integer, Integer> map = new HashMap<>();
    for (int i = 0; i < in.length; i++) map.put(in[i], i);
    return build(pre, 0, pre.length-1, in, 0, in.length-1, map);
}
TreeNode build(int[] pre, int ps, int pe, int[] in, int is, int ie, Map<Integer,Integer> map) {
    if (ps > pe) return null;
    TreeNode root = new TreeNode(pre[ps]);
    int idx = map.get(pre[ps]);
    int leftLen = idx - is;
    root.left = build(pre, ps+1, ps+leftLen, in, is, idx-1, map);
    root.right = build(pre, ps+leftLen+1, pe, in, idx+1, ie, map);
    return root;
}

6 BST 相关 / BST Problems

⑬ 验证 BST

// 中序遍历必须严格递增
boolean isValidBST(TreeNode root) {
    return validate(root, Long.MIN_VALUE, Long.MAX_VALUE);
}
boolean validate(TreeNode node, long min, long max) {
    if (node == null) return true;
    if (node.val <= min || node.val >= max) return false;
    return validate(node.left, min, node.val)
        && validate(node.right, node.val, max);
}

⑭ BST 插入节点

TreeNode insertIntoBST(TreeNode root, int val) {
    if (root == null) return new TreeNode(val);
    if (val < root.val) root.left = insertIntoBST(root.left, val);
    else root.right = insertIntoBST(root.right, val);
    return root;
}

⑮ BST 搜索区间 [k1, k2]

List<Integer> rangSearch(TreeNode root, int k1, int k2) {
    List<Integer> res = new ArrayList<>();
    inorder(root, k1, k2, res);
    return res;
}
void inorder(TreeNode node, int k1, int k2, List<Integer> res) {
    if (node == null) return;
    if (node.val > k1) inorder(node.left, k1, k2, res);
    if (node.val >= k1 && node.val <= k2) res.add(node.val);
    if (node.val < k2) inorder(node.right, k1, k2, res);
}

7 进阶题 / Advanced

⑯ 路径总和（打印所有路径）

void pathSum(TreeNode root, int target, List<Integer> path, List<List<Integer>> res) {
    if (root == null) return;
    path.add(root.val);
    if (root.left == null && root.right == null && target == root.val)
        res.add(new ArrayList<>(path));
    pathSum(root.left, target - root.val, path, res);
    pathSum(root.right, target - root.val, path, res);
    path.remove(path.size() - 1); // 回溯
}

⑰ 二叉树最长距离（直径）

int maxDist = 0;
int diameter(TreeNode root) {
    depth(root);
    return maxDist;
}
int depth(TreeNode node) {
    if (node == null) return 0;
    int l = depth(node.left), r = depth(node.right);
    maxDist = Math.max(maxDist, l + r); // 经过当前节点的最长路径
    return Math.max(l, r) + 1;
}

⑱ 不同的 BST 数量（卡特兰数）

// n 个节点能组成多少种不同的 BST？
int numTrees(int n) {
    int[] dp = new int[n + 1];
    dp[0] = dp[1] = 1;
    for (int i = 2; i <= n; i++)
        for (int j = 0; j < i; j++)
            dp[i] += dp[j] * dp[i - j - 1]; // 卡特兰数
    return dp[n];
}
🔢 卡特兰数

C(n) = C(0)×C(n-1) + C(1)×C(n-2) + ... + C(n-1)×C(0)。前几项：1, 1, 2, 5, 14, 42, 132...

8 解题模式 / Patterns

模式适用题目关键点
自顶向下递归深度、路径和、验证 BST参数传递状态
自底向上递归平衡判断、LCA、直径返回值携带信息
BFS 层序层序遍历、完全二叉树、最小深度队列 + size 分层
迭代 + 栈前/中/后序迭代、BST 迭代器模拟递归调用栈
分治构建前序+中序建树、有序数组建 BST找根 → 分左右 → 递归
🎯 面试技巧
  • 先问清楚：是普通二叉树还是 BST？有无重复值？
  • 先写递归解法（简洁），再问面试官是否需要迭代版本
  • 画图！在白板上画一棵 3~5 个节点的树，手动模拟
  • 注意边界：空树、单节点、只有左/右子树
